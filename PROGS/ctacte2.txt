SELECT resultado.cod1, SUM(resultado.saldo) as Tstock,SUM(resultado.prom2m) as T2meses,;
	   SUM(resultado.ultmes) as Tultimo, CAST(0 as n(1,0)) as Pase1,CAST(0 as n(1,0)) as Pase2,CAST(0 as n(1,0)) as Pase3,;
	   CAST(0.00 as n(10,2)) as Mysaldo;
	   FROM resultado ;
	   GROUP BY resultado.cod1 ORDER BY resultado.cod1 INTO CURSOR cabecera readwrite

	   

SELECT cabecera
GO top
SCAN 
  asd=cod1
  IF tstock=0 AND t2meses=0 AND tultimo=0
     replace pase1 WITH 0
  ELSE 
    t2m=0
    tum=0
    
    IF t2meses=0
       t2m=1
    ELSE
       t2m=t2meses
    ENDIF 
    
    IF tultimo=0
	    tum=1
    ELSE
        tum=tultimo
    ENDIF 
    
    
    IF Tstock/t2m<=3 OR Tstock/tum<=3
       replace pase1 WITH 1
    ENDIF
  ENDIF 

  SELECT resultado
  GO top
  LOCATE FOR cod1=asd AND clase="A"
  IF FOUND()
     asd1=saldo
  ENDIF 
     
  SELECT cabecera
  replace Mysaldo WITH asd1         
ENDSCAN 


SELECT cabecera
GO top
SCAN 
   asd=cod1
   IF tultimo=0
      replace pase2 WITH 0
   ELSE
      IF Mysaldo/tultimo<0.33
          replace pase2 WITH 1
      ENDIF 
   ENDIF 
   
  SELECT resultado
  GO top
  LOCATE FOR cod1=asd AND clase="Z"
  IF FOUND()
     asd1=1
  ELSE 
     asd1=0   
  ENDIF 
     
  SELECT cabecera
  replace pase3 WITH asd1                             
ENDSCAN

SELECT cabecera
GO top
SCAN 
   IF pase1=1 OR pase2=1 OR pase3=1
   
   ELSE 
   	asd1=cod1
   	DELETE 
   
   	SELECT resultado
   	GO top
   	DELETE ALL FOR cod1=asd1           
   ENDIF 
   
   SELECT cabecera

ENDSCAN
